  -- ============================================
  -- SOCIAL_ACCOUNTS (Liên kết tài khoản mạng xã hội)
  -- ============================================
  CREATE TABLE social_accounts (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
      provider        VARCHAR(20) NOT NULL,        -- 'google' | 'github' | 'facebook' | 'discord'
      provider_id     VARCHAR(255) NOT NULL,       -- ID từ provider (sub/id)
      email           VARCHAR(255),                -- Email từ provider
      display_name    VARCHAR(255),                -- Tên hiển thị từ provider
      avatar_url      VARCHAR(500),                -- Avatar từ provider
      access_token    TEXT,                         -- OAuth access token (encrypted)
      refresh_token   TEXT,                         -- OAuth refresh token (encrypted)
      token_expires_at TIMESTAMPTZ,                -- Thời hạn token
      raw_profile     JSONB,                       -- Toàn bộ profile data gốc
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW(),

      -- Mỗi provider chỉ có 1 account duy nhất
      UNIQUE(provider, provider_id),
      -- Mỗi user chỉ link 1 account/provider
      UNIQUE(user_id, provider)
  );

  CREATE INDEX idx_social_accounts_user ON social_accounts(user_id);
  CREATE INDEX idx_social_accounts_provider ON social_accounts(provider, provider_id);

  2.6 Token Encryption Service

  // src/modules/auth/services/token-encryption.service.ts
  import { Injectable } from '@nestjs/common';
  import { ConfigService } from '@nestjs/config';
  import * as crypto from 'crypto';

  @Injectable()
  export class TokenEncryptionService {
    private readonly algorithm = 'aes-256-gcm';
    private readonly key: Buffer;

    constructor(private configService: ConfigService) {
      const encryptionKey = this.configService.get<string>('socialAuth.tokenEncryptionKey');
      // Derive a 32-byte key from the config key
      this.key = crypto.scryptSync(encryptionKey, 'salt', 32);
    }

    encrypt(plainText: string): string {
      if (!plainText) return null;

      const iv = crypto.randomBytes(16);
      const cipher = crypto.createCipheriv(this.algorithm, this.key, iv);

      let encrypted = cipher.update(plainText, 'utf8', 'hex');
      encrypted += cipher.final('hex');

      const authTag = cipher.getAuthTag();

      // Format: iv:authTag:encrypted
      return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
    }

    decrypt(encryptedText: string): string {
      if (!encryptedText) return null;

      const [ivHex, authTagHex, encrypted] = encryptedText.split(':');

      const iv = Buffer.from(ivHex, 'hex');
      const authTag = Buffer.from(authTagHex, 'hex');

      const decipher = crypto.createDecipheriv(this.algorithm, this.key, iv);
      decipher.setAuthTag(authTag);

      let decrypted = decipher.update(encrypted, 'hex', 'utf8');
      decrypted += decipher.final('utf8');

      return decrypted;
    }
  }

  2.7 Google Auth Service

  // src/modules/auth/services/google-auth.service.ts
  import { Injectable, UnauthorizedException } from '@nestjs/common';
  import { ConfigService } from '@nestjs/config';
  import { OAuth2Client, TokenPayload } from 'google-auth-library';
  import { SocialProfileDto } from '../dto/social-profile.dto';
  import { SocialProvider } from '../entities/social-account.entity';

  @Injectable()
  export class GoogleAuthService {
    private oauthClient: OAuth2Client;

    constructor(private configService: ConfigService) {
      this.oauthClient = new OAuth2Client(
        this.configService.get('socialAuth.google.clientId'),
        this.configService.get('socialAuth.google.clientSecret'),
      );
    }

    /**
     * Lấy URL redirect đến Google OAuth
     */
    getAuthUrl(state?: string): string {
      const redirectUri = this.configService.get('socialAuth.google.callbackUrl');
      return this.oauthClient.generateAuthUrl({
        access_type: 'offline',
        scope: [
          'openid',
          'https://www.googleapis.com/auth/userinfo.email',
          'https://www.googleapis.com/auth/userinfo.profile',
        ],
        redirect_uri: redirectUri,
        state,
        prompt: 'consent',
      });
    }

    /**
     * Exchange authorization code cho tokens + profile
     */
    async getProfileFromCode(code: string, redirectUri?: string): Promise<SocialProfileDto> {
      try {
        const finalRedirectUri = redirectUri || this.configService.get('socialAuth.google.callbackUrl');

        const { tokens } = await this.oauthClient.getToken({
          code,
          redirect_uri: finalRedirectUri,
        });

        this.oauthClient.setCredentials(tokens);

        // Verify ID token
        const ticket = await this.oauthClient.verifyIdToken({
          idToken: tokens.id_token,
          audience: this.configService.get('socialAuth.google.clientId'),
        });

        const payload: TokenPayload = ticket.getPayload();

        if (!payload.email_verified) {
          throw new UnauthorizedException('Google email is not verified');
        }

        return {
          provider: SocialProvider.GOOGLE,
          providerId: payload.sub,
          email: payload.email,
          displayName: payload.name || payload.email.split('@')[0],
          avatarUrl: payload.picture,
          accessToken: tokens.access_token,
          refreshToken: tokens.refresh_token,
          tokenExpiresAt: tokens.expiry_date ? new Date(tokens.expiry_date) : null,
          rawProfile: payload as any,
        };
      } catch (error) {
        throw new UnauthorizedException(`Google authentication failed: ${error.message}`);
      }
    }

    /**
     * Verify Google ID Token trực tiếp (cho Google One Tap)
     */
    async getProfileFromIdToken(idToken: string): Promise<SocialProfileDto> {
      try {
        const ticket = await this.oauthClient.verifyIdToken({
          idToken,
          audience: this.configService.get('socialAuth.google.clientId'),
        });

        const payload: TokenPayload = ticket.getPayload();

        if (!payload.email_verified) {
          throw new UnauthorizedException('Google email is not verified');
        }

        return {
          provider: SocialProvider.GOOGLE,
          providerId: payload.sub,
          email: payload.email,
          displayName: payload.name || payload.email.split('@')[0],
          avatarUrl: payload.picture,
          rawProfile: payload as any,
        };
      } catch (error) {
        throw new UnauthorizedException(`Google token verification failed: ${error.message}`);
      }
    }
  }

  2.8 GitHub Auth Service

  // src/modules/auth/services/github-auth.service.ts
  import { Injectable, UnauthorizedException } from '@nestjs/common';
  import { ConfigService } from '@nestjs/config';
  import { HttpService } from '@nestjs/axios';
  import { firstValueFrom } from 'rxjs';
  import { SocialProfileDto } from '../dto/social-profile.dto';
  import { SocialProvider } from '../entities/social-account.entity';

  @Injectable()
  export class GitHubAuthService {
    constructor(
      private configService: ConfigService,
      private httpService: HttpService,
    ) {}

    /**
     * Lấy URL redirect đến GitHub OAuth
     */
    getAuthUrl(state?: string): string {
      const clientId = this.configService.get('socialAuth.github.clientId');
      const redirectUri = this.configService.get('socialAuth.github.callbackUrl');
      const scope = 'read:user user:email';

      const params = new URLSearchParams({
        client_id: clientId,
        redirect_uri: redirectUri,
        scope,
        state: state || '',
      });

      return `https://github.com/login/oauth/authorize?${params.toString()}`;
    }

    /**
     * Exchange authorization code cho tokens + profile
     */
    async getProfileFromCode(code: string, redirectUri?: string): Promise<SocialProfileDto> {
      try {
        // 1. Exchange code cho access token
        const tokenResponse = await firstValueFrom(
          this.httpService.post(
            'https://github.com/login/oauth/access_token',
            {
              client_id: this.configService.get('socialAuth.github.clientId'),
              client_secret: this.configService.get('socialAuth.github.clientSecret'),
              code,
              redirect_uri: redirectUri || this.configService.get('socialAuth.github.callbackUrl'),
            },
            {
              headers: { Accept: 'application/json' },
            },
          ),
        );

        const { access_token, token_type } = tokenResponse.data;

        if (!access_token) {
          throw new UnauthorizedException('Failed to get GitHub access token');
        }

        // 2. Lấy profile
        const profileResponse = await firstValueFrom(
          this.httpService.get('https://api.github.com/user', {
            headers: { Authorization: `${token_type} ${access_token}` },
          }),
        );

        const profile = profileResponse.data;

        // 3. Lấy email (nếu email là private)
        let email = profile.email;
        if (!email) {
          const emailsResponse = await firstValueFrom(
            this.httpService.get('https://api.github.com/user/emails', {
              headers: { Authorization: `${token_type} ${access_token}` },
            }),
          );

          const primaryEmail = emailsResponse.data.find(
            (e: any) => e.primary && e.verified,
          );
          email = primaryEmail?.email;
        }

        if (!email) {
          throw new UnauthorizedException(
            'Cannot get email from GitHub. Please make your email public or use another login method.',
          );
        }

        return {
          provider: SocialProvider.GITHUB,
          providerId: String(profile.id),
          email,
          displayName: profile.name || profile.login,
          avatarUrl: profile.avatar_url,
          accessToken: access_token,
          rawProfile: profile,
        };
      } catch (error) {
        if (error instanceof UnauthorizedException) throw error;
        throw new UnauthorizedException(`GitHub authentication failed: ${error.message}`);
      }
    }
  }

  2.9 Core Social Auth Service

  // src/modules/auth/services/social-auth.service.ts
  import {
    Injectable, UnauthorizedException, ConflictException,
    BadRequestException, Logger,
  } from '@nestjs/common';
  import { InjectRepository } from '@nestjs/typeorm';
  import { Repository, DataSource } from 'typeorm';
  import { SocialAccount, SocialProvider } from '../entities/social-account.entity';
  import { User } from '../../users/entities/user.entity';
  import { Wallet } from '../../wallet/entities/wallet.entity';
  import { GoogleAuthService } from './google-auth.service';
  import { GitHubAuthService } from './github-auth.service';
  import { TokenEncryptionService } from './token-encryption.service';
  import { AuthService } from '../auth.service';
  import { SocialProfileDto } from '../dto/social-profile.dto';

  export interface SocialAuthResult {
    accessToken: string;
    refreshToken: string;
    user: Partial<User>;
    isNewUser: boolean;
  }

  @Injectable()
  export class SocialAuthService {
    private readonly logger = new Logger(SocialAuthService.name);

    constructor(
      @InjectRepository(SocialAccount)
      private socialAccountRepo: Repository<SocialAccount>,
      @InjectRepository(User)
      private userRepo: Repository<User>,
      private dataSource: DataSource,
      private googleAuthService: GoogleAuthService,
      private gitHubAuthService: GitHubAuthService,
      private tokenEncryptionService: TokenEncryptionService,
      private authService: AuthService, // Dùng để tạo JWT tokens
    ) {}

    // ========================================
    // 1. GET AUTH URL (redirect user đến provider)
    // ========================================
    getAuthUrl(provider: SocialProvider, state?: string): string {
      switch (provider) {
        case SocialProvider.GOOGLE:
          return this.googleAuthService.getAuthUrl(state);
        case SocialProvider.GITHUB:
          return this.gitHubAuthService.getAuthUrl(state);
        default:
          throw new BadRequestException(`Provider ${provider} is not supported`);
      }
    }

    // ========================================
    // 2. SOCIAL LOGIN / REGISTER
    // ========================================
    async socialLogin(
      provider: SocialProvider,
      code: string,
      redirectUri?: string,
    ): Promise<SocialAuthResult> {
      // Step 1: Lấy profile từ provider
      const socialProfile = await this.getProfileFromProvider(provider, code, redirectUri);

      // Step 2: Tìm social account đã link chưa
      const existingSocialAccount = await this.socialAccountRepo.findOne({
        where: {
          provider: socialProfile.provider as SocialProvider,
          providerId: socialProfile.providerId,
        },
        relations: ['user'],
      });

      // Step 3: Xử lý các cases
      if (existingSocialAccount) {
        // Case A: Đã link → Login bình thường
        return this.loginExistingSocialUser(existingSocialAccount, socialProfile);
      }

      // Case B: Chưa link → Kiểm tra email đã tồn tại chưa
      const existingUser = await this.userRepo.findOne({
        where: { email: socialProfile.email },
      });

      if (existingUser) {
        // Case B1: Email đã tồn tại → Tự động link + login
        return this.linkAndLogin(existingUser, socialProfile);
      }

      // Case C: User hoàn toàn mới → Tạo account mới
      return this.createSocialUser(socialProfile);
    }

    // ========================================
    // 3. SOCIAL LOGIN VIA ID TOKEN (Google One Tap)
    // ========================================
    async socialLoginViaIdToken(
      provider: SocialProvider,
      idToken: string,
    ): Promise<SocialAuthResult> {
      let socialProfile: SocialProfileDto;

      switch (provider) {
        case SocialProvider.GOOGLE:
          socialProfile = await this.googleAuthService.getProfileFromIdToken(idToken);
          break;
        default:
          throw new BadRequestException(`ID token login not supported for ${provider}`);
      }

      // Tương tự flow socialLogin
      const existingSocialAccount = await this.socialAccountRepo.findOne({
        where: {
          provider: socialProfile.provider as SocialProvider,
          providerId: socialProfile.providerId,
        },
        relations: ['user'],
      });

      if (existingSocialAccount) {
        return this.loginExistingSocialUser(existingSocialAccount, socialProfile);
      }

      const existingUser = await this.userRepo.findOne({
        where: { email: socialProfile.email },
      });

      if (existingUser) {
        return this.linkAndLogin(existingUser, socialProfile);
      }

      return this.createSocialUser(socialProfile);
    }

    // ========================================
    // 4. LINK SOCIAL ACCOUNT (User đã đăng nhập muốn link thêm)
    // ========================================
    async linkSocialAccount(
      userId: string,
      provider: SocialProvider,
      code: string,
      redirectUri?: string,
    ): Promise<SocialAccount> {
      const socialProfile = await this.getProfileFromProvider(provider, code, redirectUri);

      // Kiểm tra provider+providerId đã được link bởi user khác chưa
      const existingLink = await this.socialAccountRepo.findOne({
        where: {
          provider: socialProfile.provider as SocialProvider,
          providerId: socialProfile.providerId,
        },
      });

      if (existingLink && existingLink.userId !== userId) {
        throw new ConflictException(
          `This ${provider} account is already linked to another user`,
        );
      }

      if (existingLink && existingLink.userId === userId) {
        // Đã link rồi → Update tokens
        return this.updateSocialAccount(existingLink, socialProfile);
      }

      // Kiểm tra user đã link provider này chưa
      const userLink = await this.socialAccountRepo.findOne({
        where: { userId, provider },
      });

      if (userLink) {
        throw new ConflictException(
          `You already have a ${provider} account linked. Unlink it first.`,
        );
      }

      // Tạo link mới
      return this.createSocialAccount(userId, socialProfile);
    }

    // ========================================
    // 5. UNLINK SOCIAL ACCOUNT
    // ========================================
    async unlinkSocialAccount(
      userId: string,
      provider: SocialProvider,
    ): Promise<void> {
      const user = await this.userRepo.findOne({ where: { id: userId } });

      // Không cho unlink nếu:
      // - User không có password (chỉ login bằng social)
      // - Và chỉ còn 1 social account
      if (!user.isPasswordSet) {
        const socialCount = await this.socialAccountRepo.count({
          where: { userId },
        });

        if (socialCount <= 1) {
          throw new BadRequestException(
            'Cannot unlink the only login method. Please set a password first.',
          );
        }
      }

      const result = await this.socialAccountRepo.delete({ userId, provider });

      if (result.affected === 0) {
        throw new BadRequestException(`No ${provider} account linked`);
      }
    }

    // ========================================
    // 6. GET LINKED ACCOUNTS
    // ========================================
    async getLinkedAccounts(userId: string): Promise<Partial<SocialAccount>[]> {
      const accounts = await this.socialAccountRepo.find({
        where: { userId },
        select: ['id', 'provider', 'email', 'displayName', 'avatarUrl', 'createdAt'],
      });

      return accounts;
    }

    private async getProfileFromProvider(
      provider: SocialProvider,
      code: string,
      redirectUri?: string,
    ): Promise<SocialProfileDto> {
      switch (provider) {
        case SocialProvider.GOOGLE:
          return this.googleAuthService.getProfileFromCode(code, redirectUri);
        case SocialProvider.GITHUB:

        default:
          throw new BadRequestException(`Provider ${provider} is not supported`);
      }
    }

    /**
     * Case A: User đã có social account → Login
     */
    private async loginExistingSocialUser(
      socialAccount: SocialAccount,
      profile: SocialProfileDto,
    ): Promise<SocialAuthResult> {
      // Update tokens nếu có mới
      await this.updateSocialAccount(socialAccount, profile);

      // Tạo JWT tokens
      const tokens = await this.authService.generateTokens(socialAccount.user);

    /**
     * Case B1: Email đã tồn tại → Auto link + login
     */
    private async linkAndLogin(
      existingUser: User,
      profile: SocialProfileDto,
    ): Promise<SocialAuthResult> {
      // Tạo social account link
      await this.createSocialAccount(existingUser.id, profile);

      // Update avatar nếu user chưa có
      if (!existingUser.avatarUrl && profile.avatarUrl) {
        await this.userRepo.update(existingUser.id, {
          avatarUrl: profile.avatarUrl,
        });
        existingUser.avatarUrl = profile.avatarUrl;
      }

      const tokens = await this.authService.generateTokens(existingUser);

      return {
        ...tokens,
        user: this.sanitizeUser(existingUser),
        isNewUser: false,
      };
    }

    /**
     * Case C: Tạo user hoàn toàn mới từ social profile
     */
    private async createSocialUser(
      profile: SocialProfileDto,
    ): Promise<SocialAuthResult> {
      const queryRunner = this.dataSource.createQueryRunner();
      await queryRunner.connect();
      await queryRunner.startTransaction();

      try {
        // 1. Tạo user mới (không có password)
        const newUser = queryRunner.manager.create(User, {
          email: profile.email,
          displayName: profile.displayName,
          avatarUrl: profile.avatarUrl,
          authProvider: profile.provider,
          isPasswordSet: false,
          passwordHash: null,
          isVerified: true, // Social login = email đã verified
          role: 'reader',
        });

        const savedUser = await queryRunner.manager.save(User, newUser);

        // 2. Tạo wallet + bonus (giống register thường)
        const wallet = queryRunner.manager.create(Wallet, {
          userId: savedUser.id,
          balance: 5000, // Bonus 5.000đ
          totalEarned: 0,
          totalSpent: 0,
        });
        await queryRunner.manager.save(Wallet, wallet);

        // 3. Tạo social account link
        const socialAccount = queryRunner.manager.create(SocialAccount, {
          userId: savedUser.id,
          provider: profile.provider as SocialProvider,
          providerId: profile.providerId,
          email: profile.email,
          displayName: profile.displayName,
          avatarUrl: profile.avatarUrl,
          accessToken: this.tokenEncryptionService.encrypt(profile.accessToken),
          refreshToken: this.tokenEncryptionService.encrypt(profile.refreshToken),
          tokenExpiresAt: profile.tokenExpiresAt,
          rawProfile: profile.rawProfile,
        });
        await queryRunner.manager.save(SocialAccount, socialAccount);

        // 4. Tạo transaction bonus (giống register thường)
        // ... (tạo transaction record cho bonus 5000đ)

        await queryRunner.commitTransaction();

        // 5. Tạo JWT tokens
        const tokens = await this.authService.generateTokens(savedUser);

        this.logger.log(
          `New social user created: ${savedUser.email} via ${profile.provider}`,
        );

        return {
          ...tokens,
          user: this.sanitizeUser(savedUser),
          isNewUser: true,
        };
      } catch (error) {
        await queryRunner.rollbackTransaction();
        this.logger.error(`Failed to create social user: ${error.message}`);
        throw error;
      } finally {
        await queryRunner.release();
      }
    }

    private async createSocialAccount(
      userId: string,
      profile: SocialProfileDto,
    ): Promise<SocialAccount> {
      const socialAccount = this.socialAccountRepo.create({
        userId,
        provider: profile.provider as SocialProvider,
        providerId: profile.providerId,
        email: profile.email,
        displayName: profile.displayName,
        avatarUrl: profile.avatarUrl,
        accessToken: this.tokenEncryptionService.encrypt(profile.accessToken),
        refreshToken: this.tokenEncryptionService.encrypt(profile.refreshToken),
        tokenExpiresAt: profile.tokenExpiresAt,
        rawProfile: profile.rawProfile,
      });

      return this.socialAccountRepo.save(socialAccount);
    }

    private async updateSocialAccount(
      existingAccount: SocialAccount,
      profile: SocialProfileDto,
    ): Promise<SocialAccount> {
      if (profile.accessToken) {
        existingAccount.accessToken = this.tokenEncryptionService.encrypt(profile.accessToken);
      }
      if (profile.refreshToken) {
        existingAccount.refreshToken = this.tokenEncryptionService.encrypt(profile.refreshToken);
      }
      if (profile.tokenExpiresAt) {
        existingAccount.tokenExpiresAt = profile.tokenExpiresAt;
      }
      existingAccount.displayName = profile.displayName;
      existingAccount.avatarUrl = profile.avatarUrl;
      existingAccount.rawProfile = profile.rawProfile;

      return this.socialAccountRepo.save(existingAccount);
    }

    private sanitizeUser(user: User): Partial<User> {
      const { passwordHash, ...sanitized } = user;
      return sanitized;
    }

          refresh_token: result.refreshToken,
          is_new_user: String(result.isNewUser),

    },

      clientId: process.env.NEXT_PUBLIC_GITHUB_CLIENT_ID!,
      authUrl: 'https://github.com/login/oauth/authorize',
      scope: 'read:user user:email',
      redirectUri: `${process.env.NEXT_PUBLIC_APP_URL}/auth/social/callback`,
    },

  /**
   * Tạo OAuth URL và mở popup/redirect
   */
  export function initiateOAuth(
    provider: SocialProvider,
    options: {
      mode?: 'redirect' | 'popup';
      state?: string;
    } = {},
  ): Window | null {
    const config = OAUTH_CONFIGS[provider];
    const { mode = 'popup', state } = options;

    // Generate random state để chống CSRF
    const oauthState = state || generateState(provider);
    sessionStorage.setItem('oauth_state', oauthState);

    const params = new URLSearchParams({
      client_id: config.clientId,
      redirect_uri: config.redirectUri,
      scope: config.scope,
      state: oauthState,
      response_type: 'code',
    });

    // Thêm params đặc biệt cho từng provider
    if (provider === 'google') {
      params.set('access_type', 'offline');
      params.set('prompt', 'consent');
    }

    const authUrl = `${config.authUrl}?${params.toString()}`;

    if (mode === 'popup') {
      const width = 500;
      const height = 600;
      const left = window.screenX + (window.outerWidth - width) / 2;
      const top = window.screenY + (window.outerHeight - height) / 2;

      return window.open(
        authUrl,
        `${provider}_oauth`,
        `width=${width},height=${height},left=${left},top=${top},toolbar=no,menubar=no`,
      );
    } else {
      window.location.href = authUrl;
      return null;
    }
  }

  /**
   * Parse callback URL parameters
   */
  export function parseOAuthCallback(searchParams: URLSearchParams): {
    code?: string;
    state?: string;
    error?: string;
    provider?: SocialProvider;
  } {
    const code = searchParams.get('code');
    const state = searchParams.get('state');
    const error = searchParams.get('error');

    // Extract provider from state
    let provider: SocialProvider | undefined;
    if (state) {
      const parts = state.split(':');
      if (parts.length >= 1) {
        provider = parts[0] as SocialProvider;
      }
    }

    return { code, state, error, provider };
  }

  /**
   * Verify state parameter (CSRF protection)
   */
  export function verifyOAuthState(state: string): boolean {
    const savedState = sessionStorage.getItem('oauth_state');
    sessionStorage.removeItem('oauth_state');
    return savedState === state;
  }

  /**
   * Get redirect URI for a provider
   */
  export function getRedirectUri(provider: SocialProvider): string {
    return OAUTH_CONFIGS[provider].redirectUri;
  }

  function generateState(provider: SocialProvider): string {
    const random = Math.random().toString(36).substring(2, 15);
    return `${provider}:${random}:${Date.now()}`;
  }

    // Google One Tap mutation
    const tokenLoginMutation = useMutation({
      mutationFn: authApi.socialTokenLogin,
      onSuccess: (response) => {
        const { accessToken, refreshToken, user, isNewUser } = response.data.data;

        setAuth(accessToken, refreshToken, user);

        if (isNewUser) {
          toast.success('🎉 Chào mừng bạn!');
          router.push('/onboarding');
        } else {
          toast.success('Đăng nhập thành công!');
          router.push('/');
        }
      },
      onError: (error: any) => {
        toast.error(error?.response?.data?.message || 'Đăng nhập thất bại');
      },
      onSettled: () => {
        setLoadingProvider(null);
      },
    });

    /**
     * Bắt đầu OAuth flow (popup mode)
     */
    const loginWithProvider = useCallback((provider: SocialProvider) => {
      setLoadingProvider(provider);

      // Mở popup
      popupRef.current = initiateOAuth(provider, { mode: 'popup' });

      if (!popupRef.current) {
        // Popup bị chặn → fallback sang redirect
        toast.info('Popup bị chặn. Đang chuyển hướng...');
        initiateOAuth(provider, { mode: 'redirect' });
        return;
      }

      // Poll popup để check khi nào đóng/có kết quả
      popupCheckInterval.current = setInterval(() => {
        if (!popupRef.current || popupRef.current.closed) {
          clearInterval(popupCheckInterval.current!);
          setLoadingProvider(null);
        }
      }, 500);
    }, []);

    /**
     * Google One Tap login
     */
    const loginWithGoogleOneTap = useCallback((idToken: string) => {
      setLoadingProvider('google');
      tokenLoginMutation.mutate({
        provider: 'google',
        idToken,
      });
    }, [tokenLoginMutation]);

    /**
     * Xử lý OAuth callback (gọi từ callback page)
     */
    const handleCallback = useCallback(
      async (provider: SocialProvider, code: string) => {
        setLoadingProvider(provider);
        socialLoginMutation.mutate({
          provider,
          code,
          redirectUri: getRedirectUri(provider),
        });
      },
      [socialLoginMutation],
    );

    // Cleanup
    useEffect(() => {
      return () => {
        if (popupCheckInterval.current) {
          clearInterval(popupCheckInterval.current);
        }
      };
    }, []);

    // Lắng nghe message từ popup
    useEffect(() => {
      const handleMessage = (event: MessageEvent) => {
        if (event.origin !== window.location.origin) return;

        if (event.data?.type === 'SOCIAL_AUTH_CALLBACK') {
          const { provider, code, error } = event.data;

          if (popupRef.current) {
            popupRef.current.close();
          }

          if (error) {
            toast.error(`Đăng nhập ${provider} thất bại: ${error}`);
            setLoadingProvider(null);
            return;
          }

          if (code && provider) {
            handleCallback(provider, code);
          }
        }
      };

      window.addEventListener('message', handleMessage);
      return () => window.removeEventListener('message', handleMessage);
    }, [handleCallback]);

    return {
      loginWithProvider,
      loginWithGoogleOneTap,
      handleCallback,
      loadingProvider,
      isLoading: socialLoginMutation.isPending || tokenLoginMutation.isPending,
    };
  }

  // src/app/auth/social/callback/page.tsx
  'use client';

  import { useEffect, useRef, useState } from 'react';
  import { useSearchParams, useRouter } from 'next/navigation';
  import { Loader2, CheckCircle, XCircle } from 'lucide-react';
  import { useSocialAuth } from '@/hooks/useSocialAuth';
  import { useAuthStore } from '@/stores/authStore';
  import {
    parseOAuthCallback,
    verifyOAuthState,
  } from '@/lib/social-auth';

  type CallbackStatus = 'processing' | 'success' | 'error';

  export default function SocialCallbackPage() {
    const searchParams = useSearchParams();
    const router = useRouter();
    const { handleCallback } = useSocialAuth();
    const { setAuth } = useAuthStore();
    const [status, setStatus] = useState<CallbackStatus>('processing');
    const [errorMessage, setErrorMessage] = useState('');
    const processed = useRef(false);

    useEffect(() => {
      if (processed.current) return;
      processed.current = true;

      // ============================================
      // Case 1: Server-side redirect (tokens in URL)
      // ============================================
      const accessToken = searchParams.get('access_token');
      const refreshToken = searchParams.get('refresh_token');

      if (accessToken && refreshToken) {
        const isNewUser = searchParams.get('is_new_user') === 'true';

        setAuth(accessToken, refreshToken, null); // User sẽ được fetch sau
        setStatus('success');

        // Nếu đang trong popup → gửi message về parent window
        if (window.opener) {
          window.opener.postMessage(
            {
              type: 'SOCIAL_AUTH_SUCCESS',
              accessToken,
              refreshToken,
              isNewUser,
            },
            window.location.origin,
          );
          window.close();
          return;
        }

        // Redirect
        setTimeout(() => {
          router.push(isNewUser ? '/onboarding' : '/');
        }, 1000);

        return;
      }

      // ============================================
      // Case 2: Client-side OAuth (code in URL)
      // ============================================
      const { code, state, error, provider } = parseOAuthCallback(
        new URLSearchParams(searchParams.toString()),
      );

      if (error) {
        setStatus('error');
        setErrorMessage(error);

        // Gửi error về popup parent
        if (window.opener) {
          window.opener.postMessage(
            { type: 'SOCIAL_AUTH_CALLBACK', error, provider },
            window.location.origin,
          );
          window.close();
        }
        return;
      }

      if (!code || !provider) {
        setStatus('error');
        setErrorMessage('Invalid callback parameters');
        return;
      }

      // Verify state (CSRF protection)
      if (state && !verifyOAuthState(state)) {
        setStatus('error');
        setErrorMessage('Invalid state parameter. Please try again.');
        return;
      }

      // Nếu đang trong popup → gửi code về parent window xử lý
      if (window.opener) {
        window.opener.postMessage(
          { type: 'SOCIAL_AUTH_CALLBACK', provider, code },
          window.location.origin,
        );
        window.close();
        return;
      }

      // Nếu không phải popup (redirect mode) → xử lý trực tiếp
      handleCallback(provider, code);
    }, [searchParams, handleCallback, router, setAuth]);

    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-50 dark:bg-gray-900">
        <div className="text-center space-y-4 p-8">
          {status === 'processing' && (
            <>
              <Loader2 className="w-12 h-12 animate-spin text-blue-500 mx-auto" />
              <h2 className="text-xl font-semibold text-gray-700 dark:text-gray-200">
                Đang xử lý đăng nhập...
              </h2>
              <p className="text-gray-500">
                Vui lòng chờ trong giây lát
              </p>
            </>
          )}

          {status === 'success' && (
            <>
              <CheckCircle className="w-12 h-12 text-green-500 mx-auto" />
              <h2 className="text-xl font-semibold text-gray-700 dark:text-gray-200">
                Đăng nhập thành công!
              </h2>
              <p className="text-gray-500">
                Đang chuyển hướng...
              </p>
            </>
          )}

          {status === 'error' && (
            <>
              <XCircle className="w-12 h-12 text-red-500 mx-auto" />
              <h2 className="text-xl font-semibold text-gray-700 dark:text-gray-200">
                Đăng nhập thất bại

                           hover:bg-blue-600 transition-colors"
              >
                Quay lại đăng nhập
              </button>
            </>
          )}
        </div>
      </div>
    );
  }

  export function GoogleOneTap() {
    const { isAuthenticated } = useAuthStore();
    const { loginWithGoogleOneTap } = useSocialAuth();

    const handleCredentialResponse = useCallback(
      (response: any) => {
        if (response.credential) {
          loginWithGoogleOneTap(response.credential);
        }
      },
      [loginWithGoogleOneTap],
    );

    useEffect(() => {
      // Không hiện One Tap nếu đã đăng nhập
      if (isAuthenticated) return;

      // Load Google Identity Services SDK
      const script = document.createElement('script');
      script.src = 'https://accounts.google.com/gsi/client';
      script.async = true;
      script.defer = true;

      script.onload = () => {
        if (window.google) {
          window.google.accounts.id.initialize({
            client_id: process.env.NEXT_PUBLIC_GOOGLE_CLIENT_ID,
            callback: handleCredentialResponse,
            auto_select: false, // Không tự động chọn account
            cancel_on_tap_outside: true,
          });

          // Hiển thị One Tap popup
          window.google.accounts.id.prompt((notification: any) => {
            if (notification.isNotDisplayed() || notification.isSkippedMoment()) {
              // One Tap bị ẩn (user đã dismiss trước đó, etc.)
              console.log('Google One Tap not displayed:', notification.getNotDisplayedReason?.());
            }
          });
        }
      };

      document.body.appendChild(script);

      return () => {
        // Cleanup
        if (window.google) {
          window.google.accounts.id.cancel();
        }
        document.body.removeChild(script);
      };
    }, [isAuthenticated, handleCredentialResponse]);

    // Component không render gì - One Tap tự hiện popup
    return null;

    const handleLinkMessage = useCallback(
      (event: MessageEvent) => {
        if (event.origin !== window.location.origin) return;
        if (event.data?.type === 'SOCIAL_AUTH_CALLBACK') {
          const { provider, code, error } = event.data;
          if (error) {
            toast.error(`Liên kết ${provider} thất bại: ${error}`);
            return;
          }
          if (code && provider) {
            linkMutation.mutate({ provider, code });
          }
        }
      },
      [linkMutation],
    );

    useEffect(() => {
      window.addEventListener('message', handleLinkMessage);
      return () => window.removeEventListener('message', handleLinkMessage);
    }, [handleLinkMessage]);

    const handleLink = (provider: SocialProvider) => {
      initiateOAuth(provider, { mode: 'popup' });
    };

    const isLinked = (providerId: SocialProvider) =>
      linkedAccounts?.some((a) => a.provider === providerId);

    const getLinkedAccount = (providerId: SocialProvider) =>
      linkedAccounts?.find((a) => a.provider === providerId);

    if (isLoading) {
      return (
        <div className="flex justify-center py-8">
          <Loader2 className="w-6 h-6 animate-spin text-gray-400" />
        </div>
      );
    }

    return (
      <Card className="p-6">
        <h3 className="text-lg font-semibold mb-4">Tài khoản liên kết</h3>
        <p className="text-sm text-gray-500 mb-6">
          Liên kết tài khoản mạng xã hội để đăng nhập nhanh hơn
        </p>

        <div className="space-y-4">
          {PROVIDERS.map((provider) => {
            const linked = isLinked(provider.id);
            const account = getLinkedAccount(provider.id);

            return (
              <div
                key={provider.id}
                className={`flex items-center justify-between p-4 rounded-lg border
                           ${linked ? provider.color : 'border-gray-200 dark:border-gray-700'}`}
              >
                <div className="flex items-center gap-3">
                  <div className="w-10 h-10 flex items-center justify-center rounded-full
                                  bg-white dark:bg-gray-700 shadow-sm">
                    {provider.icon}
                  </div>
                  <div>
                    <p className="font-medium">{provider.name}</p>
                    {linked && account ? (
                      <p className="text-sm text-gray-500">{account.email}</p>
                    ) : (
                      <p className="text-sm text-gray-400">Chưa liên kết</p>
                    )}
                  </div>
                </div>

                {linked ? (
                  <Button
                    variant="ghost"
                    size="sm"
                    className="text-red-500 hover:text-red-600 hover:bg-red-50"
                    onClick={() => unlinkMutation.mutate(provider.id)}
                    disabled={unlinkMutation.isPending}
                  >
                    {unlinkMutation.isPending ? (
                      <Loader2 className="w-4 h-4 animate-spin mr-1" />
                    ) : (
                      <Unlink className="w-4 h-4 mr-1" />
                    )}
                    Gỡ liên kết
                  </Button>
                ) : (
                  <Button
                    variant="outline"
                    size="sm"
                    onClick={() => handleLink(provider.id)}
                    disabled={linkMutation.isPending}
                  >
                    {linkMutation.isPending ? (
                      <Loader2 className="w-4 h-4 animate-spin mr-1" />
                    ) : (
                      <Link2 className="w-4 h-4 mr-1" />
                    )}
                    Liên kết
                  </Button>
                )}
              </div>
            );
          })}
        </div>
      </Card>
    );
  }

  4.3 Security Considerations

  ┌─────────────────────────────────────────────────────────────────┐
  │                      SECURITY MEASURES                           │
  ├─────────────────────────────────────────────────────────────────┤
  │                                                                  │
  │  1. CSRF Protection                                              │
  │     - state parameter with random + timestamp                    │
  │     - Verify state on callback                                   │
  │                                                                  │
  │  2. Token Security                                               │
  │     - OAuth tokens encrypted at rest (AES-256-GCM)              │
  │     - JWT tokens in httpOnly cookies (optional)                  │
  │     - Short-lived access tokens (15m)                            │
  │                                                                  │
  │  3. Popup vs Redirect                                            │
  │     - Popup: tokens never in URL, postMessage same-origin only   │
  │     - Redirect: tokens in URL briefly, cleared after processing  │
  │                                                                  │
  │  4. Account Takeover Prevention                                  │
  │     - Email must be verified by provider                         │
  │     - Auto-link only if emails match exactly                     │
  │     - Log all social auth events                                 │
  │                                                                  │
  │  5. Unlink Protection                                            │
  │     - Cannot unlink last login method                            │
  │     - Must set password before unlinking all social accounts     │
  │                                                                  │
  │  6. Rate Limiting                                                │
  │     - Max 10 social login attempts per IP per minute             │
  │     - Max 5 failed attempts → temporary block                    │
  │                                                                  │
  └─────────────────────────────────────────────────────────────────┘

   // app/auth/social/callback/page.tsx
   export default function SocialCallbackPage() {
     useEffect(() => {
       // 1. Lấy code và state từ URL
       const params = new URLSearchParams(window.location.search);
       const code = params.get('code');     // "4/0AbC...xYz"
       const state = params.get('state');   // "a1b2c3d4-e5f6-..."
       const error = params.get('error');   // null (nếu thành công)

       // 2. Kiểm tra có lỗi không (user cancel, etc.)
       if (error) {
         window.opener?.postMessage(
           { type: 'SOCIAL_AUTH_ERROR', error },
           window.location.origin
         );
         window.close();
         return;
       }

       // 3. Gửi code về trang gốc (trang Login) qua postMessage
       if (window.opener) {
         // ← POPUP MODE: gửi message cho trang cha
         window.opener.postMessage(
           {
             type: 'SOCIAL_AUTH_CALLBACK',
             payload: { code, state, provider: 'google' }
           },
           window.location.origin  // Chỉ gửi cho same-origin (bảo mật)
         );
         window.close(); // Đóng popup
       } else {
         // ← REDIRECT MODE (fallback): tự xử lý luôn tại đây
         handleSocialLogin(code, state, 'google');
       }
     }, []);

     return <div>Đang xử lý...</div>;
   }

   Giải thích:
   - window.opener = tham chiếu đến trang Login (trang mở popup)
   - postMessage = cách an toàn để 2 window giao tiếp
   - Popup tự đóng sau khi gửi message

   // Trang Login - handleMessage function
   const handleMessage = async (event: MessageEvent) => {
     // 1. Kiểm tra bảo mật: message phải từ cùng origin
     if (event.origin !== window.location.origin) return;

     // 2. Kiểm tra đúng loại message
     if (event.data?.type !== 'SOCIAL_AUTH_CALLBACK') return;

     const { code, state, provider } = event.data.payload;

     // 3. ⚡ VERIFY STATE (chống CSRF)
     const savedState = sessionStorage.getItem('oauth_state');
     if (state !== savedState) {
       toast.error('Xác thực không hợp lệ. Vui lòng thử lại.');
       return;
       // → Nếu state không khớp = có người đang cố tấn công
     }
     sessionStorage.removeItem('oauth_state'); // Dùng 1 lần

     // 4. 🚀 GỬI CODE LÊN BACKEND
     try {
       const response = await authApi.socialLogin({
         provider: 'google',
         code: code,          // "4/0AbC...xYz"
         redirectUri: `${window.location.origin}/auth/social/callback`,
       });

       // 5. Nhận tokens từ backend
       const { accessToken, refreshToken, user, isNewUser } = response.data.data;

       // 6. Lưu tokens
       useAuthStore.getState().setAuth(accessToken, refreshToken, user);

       // 7. Redirect
       if (isNewUser) {
         router.push('/onboarding');  // User mới → trang welcome
       } else {
         router.push('/');            // User cũ → trang chủ
       }
     } catch (error) {
       toast.error('Đăng nhập thất bại');
     }
   };

   Giải thích:
   - Verify state: so sánh state nhận được với state đã lưu → phải khớp
   - Frontend gửi code lên backend (KHÔNG TỰ exchange code)
   - Lý do: backend mới có client_secret để exchange code

   // google-auth.service.ts
   async getProfileFromCode(code: string, redirectUri: string) {
     //
     // Backend gửi request đến Google:
     //
     // POST https://oauth2.googleapis.com/token
     // {
     //   code: "4/0AbC...xYz",
     //   client_id: "123456.apps.googleusercontent.com",
     //   client_secret: "GOCSPX-xxxxx",          ← CHỈ BACKEND MỚI CÓ
     //   redirect_uri: "http://localhost:3001/auth/social/callback",
     //   grant_type: "authorization_code"
     // }
     //
     const { tokens } = await this.oauthClient.getToken({ code, redirect_uri: redirectUri });
     // tokens = {
     //   access_token: "ya29.xxx...",
     //   refresh_token: "1//xxx...",
     //   id_token: "eyJhbGciOiJS...",    ← JWT chứa thông tin user
     //   expiry_date: 1679999999999
     // }

     //
     // Verify id_token (JWT từ Google)
     //
     const ticket = await this.oauthClient.verifyIdToken({
       idToken: tokens.id_token,
       audience: this.clientId,
     });

     const payload = ticket.getPayload();
     // payload = {
     //   sub: "109876543210",          ← Google User ID (unique, không đổi)
     //   email: "user@gmail.com",
     //   email_verified: true,
     //   name: "Nguyễn Văn A",
     //   picture: "https://lh3.googleusercontent.com/...",
     //   ...
     // }

     // Kiểm tra email đã verified
     if (!payload.email_verified) {
       throw new UnauthorizedException('Email chưa được xác minh');
     }

     return {
       provider: 'google',
       providerId: payload.sub,       // "109876543210"
       email: payload.email,          // "user@gmail.com"
       displayName: payload.name,     // "Nguyễn Văn A"
       avatarUrl: payload.picture,
       accessToken: tokens.access_token,
       refreshToken: tokens.refresh_token,
       tokenExpiresAt: new Date(tokens.expiry_date),
       rawProfile: payload,
     };
   }

     ✅ Hết hạn sau ~5 phút

     ✅ Cần client_secret mới exchange được

     - Không có client_secret
     - Code đã hết hạn hoặc đã dùng rồi

   // auth.service.ts
   async generateTokens(user: User) {
     // Access Token (15 phút)
     const accessToken = this.jwtService.sign(
       {
         sub: user.id,              // "u1"
         email: user.email,         // "a@gmail.com"
         role: user.role,           // "reader"
       },
       {
         secret: this.configService.get('JWT_ACCESS_SECRET'),
         expiresIn: '15m',
       },
     );
     // accessToken = "eyJhbGciOiJIUzI1NiIs..."

     // Refresh Token (7 ngày)
     const refreshToken = this.jwtService.sign(
       { sub: user.id },
       {
         secret: this.configService.get('JWT_REFRESH_SECRET'),
         expiresIn: '7d',
       },
     );

     // Lưu refresh token hash vào Redis (để revoke được)
     await this.redis.set(
       `refresh_token:${user.id}`,
       this.hashToken(refreshToken),
       'EX',
       7 * 24 * 60 * 60, // 7 ngày
     );

     return { accessToken, refreshToken };
   }

   Tại sao dùng JWT riêng, không dùng Google token?
   Google Access Token:
     ❌ Chỉ dùng để gọi Google API
     ❌ Không chứa thông tin role/permission của app mình
     ❌ Expiry do Google quyết định

   JWT của app mình:
     ✅ Chứa user.id, role, email
     ✅ Dùng cho TẤT CẢ API của app
     ✅ Mình kiểm soát expiry
     ✅ Cùng format với user đăng nhập email/password

   → Sau khi social login, mọi thứ đều dùng JWT của app
   → Không phân biệt user login bằng cách nào

   🔐 GIẢI THÍCH CÁC CƠ CHẾ BẢO MẬT

   1. State Parameter (Chống CSRF)

   CSRF Attack scenario (không có state):

   1. Hacker tạo link:
      https://accounts.google.com/o/oauth2/auth?
        client_id=YOUR_APP
        &redirect_uri=http://localhost:3001/auth/social/callback
        &response_type=code

   2. Hacker gửi link cho victim qua email/chat

   3. Victim click → Đăng nhập Google → code gửi về app

   4. ❌ App không biết request này do USER bắt đầu hay do HACKER gửi link

   VỚI STATE:

   1. User click button → Tạo state random → Lưu sessionStorage
      state = "a1b2c3d4"
      sessionStorage['oauth_state'] = "a1b2c3d4"

   2. Gửi state trong URL: &state=a1b2c3d4

   3. Google redirect về: ?code=xxx&state=a1b2c3d4

   4. App kiểm tra: state nhận được === sessionStorage['oauth_state']?
      ✅ Khớp → Chính user này bắt đầu request
      ❌ Không khớp → Ai đó đang cố tấn công → Reject

   5. Hacker không thể biết state random → Không thể giả mạo

   2. PostMessage Security

   // Popup gửi message:
   window.opener.postMessage(data, 'http://localhost:3001');
                                     ↑
                             CHỈ GỬI CHO ORIGIN NÀY

   // Trang Login nhận message:
   window.addEventListener('message', (event) => {
     if (event.origin !== 'http://localhost:3001') return;
     //                     ↑
     //            CHỈ CHẤP NHẬN TỪ ORIGIN NÀY
   });

   → Trang web khác KHÔNG THỂ đọc được message này
   → Trang web khác KHÔNG THỂ gửi message giả mạo (sẽ bị origin check chặn)

   3. Code Exchange an toàn

   TẠI SAO KHÔNG EXCHANGE CODE Ở FRONTEND?

   Frontend (public):
     ❌ client_secret nằm trong JS → ai cũng đọc được
     ❌ Hacker lấy client_secret → giả mạo app của bạn

   Backend (private):
     ✅ client_secret nằm trong .env → không ai thấy
     ✅ Chỉ backend mới exchange được code

   FLOW AN TOÀN:
     Frontend có: code (tạm thời, 1 lần)
     Backend có:  code + client_secret → exchange → tokens + profile
     Frontend nhận: JWT của app (không phải Google tokens)

   4. Token Encryption

   Google OAuth tokens lưu trong database:

   KHÔNG MÃ HÓA (❌):
     access_token: "ya29.a0AbC..."
     → Nếu database bị hack → hacker có Google token → truy cập Google API

   CÓ MÃ HÓA (✅):
     access_token: "7f8a9b2c:3d4e5f:a1b2c3d4e5f6..."  (encrypted)
     → Nếu database bị hack → hacker chỉ thấy chuỗi vô nghĩa
     → Cần encryption key (chỉ backend có) mới decrypt được

   Headers: { Authorization: "Bearer eyJhbG..." }  ← Đã login
   Body: { provider: "google", code: "4/0AbC..." }

   Backend:
     1. Verify JWT → biết user hiện tại là ai
     2. Exchange code → lấy Google profile
     3. Kiểm tra Google account chưa link user khác
     4. Tạo social_accounts { user_id: current_user, provider: "google", ... }
     5. Trả về thành công

   → Lần sau user có thể login bằng Google hoặc email/password

   Unlink tài khoản

   POST /api/v1/auth/social/unlink
   Headers: { Authorization: "Bearer eyJhbG..." }
   Body: { provider: "github" }

   Backend:
     1. Verify JWT → user hiện tại
     2. ⚠️  KIỂM TRA: user còn cách đăng nhập nào khác không?
        - isPasswordSet = true?  → OK, có thể login bằng password
        - Còn social account khác? → OK, có thể login bằng cái khác
        - KHÔNG CÒN GÌ? → ❌ REJECT! Không cho gỡ

     3. Nếu OK → Xóa social_accounts { user_id, provider: "github" }

     const otherSocials = await socialAccountRepo.count({
