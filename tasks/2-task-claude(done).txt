  🛠️  TECH STACK

  Backend:        NestJS (Node.js + TypeScript)
  Database:       PostgreSQL + pgvector extension
  Cache:          Redis
  Auth:           JWT + Refresh Token
  Payment:        VNPay / MoMo SDK
  AI:             Google Gemini API (hoặc OpenAI)
  Embedding:      text-embedding-004 (Google)
  Queue:          BullMQ (Redis-based)

  🗄️  PHASE 1: DATABASE SCHEMA DESIGN (Tuần 1)

  Nguyên tắc quan trọng:

  - Tiền → dùng DECIMAL hoặc BIGINT (đơn vị đồng), KHÔNG BAO GIỜ dùng FLOAT
  - Mọi thao tác liên quan tiền → Database Transaction + Row Lock
  - Mọi giao dịch đều phải ghi log vào bảng transactions

  Danh sách Tables:

  -- ============================================
  -- 1. USERS
  -- ============================================
  CREATE TABLE users (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      email           VARCHAR(255) UNIQUE NOT NULL,
      password_hash   VARCHAR(255) NOT NULL,
      display_name    VARCHAR(100) NOT NULL,
      avatar_url      VARCHAR(500),
      bio             TEXT,
      role            VARCHAR(20) DEFAULT 'reader',  -- 'reader' | 'author' | 'admin'
      is_verified     BOOLEAN DEFAULT FALSE,
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW()
  );

  -- ============================================
  -- 2. WALLETS (Ví nội bộ)
  -- ============================================
  CREATE TABLE wallets (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id         UUID UNIQUE NOT NULL REFERENCES users(id),
      balance         BIGINT DEFAULT 0 CHECK (balance >= 0),  -- đơn vị: đồng
      total_earned    BIGINT DEFAULT 0,  -- tổng tiền đã kiếm được
      total_spent     BIGINT DEFAULT 0,  -- tổng tiền đã chi
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW()
  );

  -- Ví hệ thống (admin wallet) cũng là 1 row trong bảng này

  -- ============================================
  -- 3. TRANSACTIONS (Lịch sử giao dịch - KHÔNG BAO GIỜ XÓA)
  -- ============================================
  CREATE TYPE transaction_type AS ENUM (
      'deposit',              -- nạp tiền
      'withdrawal',           -- rút tiền
      'question_to_author',   -- hỏi tác giả
      'question_to_ai',       -- hỏi AI
      'refund',               -- hoàn tiền (48h timeout)
      'withdrawal_fee',       -- phí rút tiền
      'bonus'                 -- thưởng (seed 5.000đ cho user mới)
  );

  CREATE TYPE transaction_status AS ENUM (
      'pending',
      'completed',
      'failed',
      'refunded'
  );

  CREATE TABLE transactions (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      sender_id       UUID REFERENCES users(id),      -- NULL nếu là deposit
      receiver_id     UUID REFERENCES users(id),       -- NULL nếu là withdrawal
      amount          BIGINT NOT NULL CHECK (amount > 0),
      type            transaction_type NOT NULL,
      status          transaction_status DEFAULT 'pending',
      reference_id    UUID,           -- link đến question_id hoặc deposit_id
      reference_type  VARCHAR(50),    -- 'question' | 'deposit' | 'withdrawal'
      metadata        JSONB,          -- dữ liệu bổ sung
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW()
  );

  CREATE INDEX idx_transactions_sender ON transactions(sender_id);
  CREATE INDEX idx_transactions_receiver ON transactions(receiver_id);
  CREATE INDEX idx_transactions_status ON transactions(status);
  CREATE INDEX idx_transactions_created ON transactions(created_at);

  -- ============================================
  -- 4. POSTS (Bài viết)
  -- ============================================
  CREATE TABLE posts (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      author_id       UUID NOT NULL REFERENCES users(id),
      title           VARCHAR(500) NOT NULL,
      slug            VARCHAR(600) UNIQUE NOT NULL,
      content         JSONB NOT NULL,         -- Block JSON từ Editor.js / TipTap
      content_plain   TEXT,                   -- Plain text để full-text search
      excerpt         VARCHAR(500),           -- Tóm tắt
      cover_image     VARCHAR(500),
      category_id     UUID REFERENCES categories(id),
      status          VARCHAR(20) DEFAULT 'draft',  -- 'draft' | 'published' | 'archived'
      view_count      BIGINT DEFAULT 0,
      published_at    TIMESTAMPTZ,
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW()
  );

  CREATE INDEX idx_posts_author ON posts(author_id);
  CREATE INDEX idx_posts_status ON posts(status);
  CREATE INDEX idx_posts_published ON posts(published_at DESC);

  -- ============================================
  -- 5. CATEGORIES
  -- ============================================
  CREATE TABLE categories (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      name            VARCHAR(100) NOT NULL,
      slug            VARCHAR(120) UNIQUE NOT NULL,
      description     TEXT,
      created_at      TIMESTAMPTZ DEFAULT NOW()
  );

  -- ============================================
  -- 6. TAGS & POST_TAGS
  -- ============================================
  CREATE TABLE tags (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      name            VARCHAR(50) UNIQUE NOT NULL,
      slug            VARCHAR(60) UNIQUE NOT NULL
  );

  CREATE TABLE post_tags (
      post_id         UUID REFERENCES posts(id) ON DELETE CASCADE,
      tag_id          UUID REFERENCES tags(id) ON DELETE CASCADE,
      PRIMARY KEY (post_id, tag_id)
  );

  -- ============================================
  -- 7. COMMENTS (Bình luận miễn phí)
  -- ============================================
  CREATE TABLE comments (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      post_id         UUID NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
      user_id         UUID NOT NULL REFERENCES users(id),
      parent_id       UUID REFERENCES comments(id),   -- reply comment
      content         TEXT NOT NULL,
      is_hidden       BOOLEAN DEFAULT FALSE,
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW()
  );

  CREATE INDEX idx_comments_post ON comments(post_id);

  -- ============================================
  -- 8. QUESTIONS (Câu hỏi có phí - CORE FEATURE)
  -- ============================================
  CREATE TYPE question_target AS ENUM ('author', 'ai');
  CREATE TYPE question_status AS ENUM (
      'pending',      -- chờ trả lời
      'answered',     -- đã trả lời
      'refunded',     -- đã hoàn tiền (quá 48h)
      'expired'       -- hết hạn
  );

  CREATE TABLE questions (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      post_id         UUID NOT NULL REFERENCES posts(id),
      asker_id        UUID NOT NULL REFERENCES users(id),
      target          question_target NOT NULL,   -- 'author' hoặc 'ai'
      content         TEXT NOT NULL,
      answer          TEXT,                        -- câu trả lời
      answered_by     UUID REFERENCES users(id),  -- NULL nếu AI trả lời
      fee             BIGINT NOT NULL DEFAULT 1000,
      transaction_id  UUID REFERENCES transactions(id),
      status          question_status DEFAULT 'pending',
      is_highlighted  BOOLEAN DEFAULT TRUE,       -- hiển thị nổi bật
      deadline_at     TIMESTAMPTZ,                -- thời hạn 48h
      answered_at     TIMESTAMPTZ,
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      updated_at      TIMESTAMPTZ DEFAULT NOW()
  );

  CREATE INDEX idx_questions_post ON questions(post_id);
  CREATE INDEX idx_questions_status ON questions(status);
  CREATE INDEX idx_questions_deadline ON questions(deadline_at)
      WHERE status = 'pending';

  -- ============================================
  -- 9. DEPOSITS (Lịch sử nạp tiền)
  -- ============================================
  CREATE TABLE deposits (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id         UUID NOT NULL REFERENCES users(id),
      amount          BIGINT NOT NULL CHECK (amount >= 50000), -- tối thiểu 50.000đ
      payment_method  VARCHAR(50),    -- 'vnpay' | 'momo'
      payment_ref     VARCHAR(255),   -- mã giao dịch từ cổng thanh toán
      status          VARCHAR(20) DEFAULT 'pending', -- 'pending'|'completed'|'failed'
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      completed_at    TIMESTAMPTZ
  );

  -- ============================================
  -- 10. WITHDRAWALS (Lịch sử rút tiền)
  -- ============================================
  CREATE TABLE withdrawals (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id         UUID NOT NULL REFERENCES users(id),
      amount          BIGINT NOT NULL CHECK (amount >= 100000), -- tối thiểu 100.000đ
      fee_amount      BIGINT DEFAULT 0,       -- phí rút (5-10%)
      bank_name       VARCHAR(100),
      bank_account    VARCHAR(50),
      bank_holder     VARCHAR(100),
      status          VARCHAR(20) DEFAULT 'pending',
      approved_by     UUID REFERENCES users(id),
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      completed_at    TIMESTAMPTZ
  );

  -- ============================================
  -- 11. POST_EMBEDDINGS (RAG AI - Vector Storage)
  -- ============================================
  CREATE EXTENSION IF NOT EXISTS vector;

  CREATE TABLE post_embeddings (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      post_id         UUID NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
      chunk_index     INT NOT NULL,           -- thứ tự đoạn
      chunk_text      TEXT NOT NULL,           -- nội dung đoạn văn
      embedding       vector(768) NOT NULL,   -- vector 768 chiều (Google embedding)
      metadata        JSONB,
      created_at      TIMESTAMPTZ DEFAULT NOW()
  );

  CREATE INDEX idx_embeddings_post ON post_embeddings(post_id);

  -- HNSW index cho tìm kiếm vector nhanh
  CREATE INDEX idx_embeddings_vector ON post_embeddings
      USING hnsw (embedding vector_cosine_ops);

  -- ============================================
  -- 12. AUTHOR_DOCUMENTS (Tài liệu ẩn cho RAG)
  -- ============================================
  CREATE TABLE author_documents (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      author_id       UUID NOT NULL REFERENCES users(id),
      post_id         UUID REFERENCES posts(id),  -- NULL = tài liệu chung của tác giả
      file_url        VARCHAR(500),
      file_name       VARCHAR(255),
      content_plain   TEXT,           -- text đã extract từ PDF
      is_processed    BOOLEAN DEFAULT FALSE,
      created_at      TIMESTAMPTZ DEFAULT NOW()
  );

  -- Embeddings cho tài liệu ẩn
  CREATE TABLE document_embeddings (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      document_id     UUID NOT NULL REFERENCES author_documents(id) ON DELETE CASCADE,
      author_id       UUID NOT NULL REFERENCES users(id),
      chunk_index     INT NOT NULL,
      chunk_text      TEXT NOT NULL,
      embedding       vector(768) NOT NULL,
      created_at      TIMESTAMPTZ DEFAULT NOW()
  );

  CREATE INDEX idx_doc_embeddings_author ON document_embeddings(author_id);
  CREATE INDEX idx_doc_embeddings_vector ON document_embeddings
      USING hnsw (embedding vector_cosine_ops);

  -- ============================================
  -- 13. REACTIONS (Like/Heart cho câu hỏi hay)
  -- ============================================
  CREATE TABLE reactions (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id         UUID NOT NULL REFERENCES users(id),
      target_id       UUID NOT NULL,          -- question_id hoặc comment_id
      target_type     VARCHAR(20) NOT NULL,   -- 'question' | 'comment'
      type            VARCHAR(20) DEFAULT 'like',
      created_at      TIMESTAMPTZ DEFAULT NOW(),
      UNIQUE(user_id, target_id, target_type)
  );

  💰 PHASE 4: WALLET ENGINE - LOGIC CHI TIẾT (Tuần 3-4)
                                                                                                                                                                           ─
  Flow 1: User đặt Question cho Tác giả

  ┌─────────┐     ┌──────────┐     ┌──────────┐     ┌──────────┐
  │  User    │────▶│  API     │────▶│  Wallet  │────▶│  DB      │
  │  bấm    │     │  Server  │     │  Service │     │ Transaction│
  │  "Hỏi"  │     │          │     │          │     │          │
  └─────────┘     └──────────┘     └──────────┘     └──────────┘

  // questions.service.ts - PSEUDO CODE
  async createQuestion(userId: string, postId: string, dto: CreateQuestionDto) {
    const queryRunner = this.dataSource.createQueryRunner();
    await queryRunner.connect();
    await queryRunner.startTransaction('SERIALIZABLE');

    try {
      // 1. Lock ví user
      const userWallet = await queryRunner.query(
        `SELECT * FROM wallets WHERE user_id = $1 FOR UPDATE`,
        [userId]
      );

      // 2. Kiểm tra số dư
      if (userWallet.balance < 1000) {
        throw new BadRequestException('Số dư không đủ. Vui lòng nạp thêm.');
      }

      // 3. Xác định target
      const post = await this.postsRepo.findOne({ where: { id: postId } });
      const isAI = dto.target === 'ai';  // hoặc detect @AI trong content

      // 4. Trừ tiền user
      await queryRunner.query(
        `UPDATE wallets SET balance = balance - 1000,
         total_spent = total_spent + 1000,
         updated_at = NOW()
         WHERE user_id = $1`,
        [userId]
      );

      // 5. Cộng tiền receiver
      const receiverId = isAI ? SYSTEM_ADMIN_ID : post.authorId;
      await queryRunner.query(
        `UPDATE wallets SET balance = balance + 1000,
         total_earned = total_earned + 1000,
         updated_at = NOW()
         WHERE user_id = $1`,
        [receiverId]
      );

      // 6. Ghi transaction log
      const transaction = await queryRunner.query(
        `INSERT INTO transactions
         (sender_id, receiver_id, amount, type, status, reference_type)
         VALUES ($1, $2, 1000, $3, 'completed', 'question')
         RETURNING id`,
        [userId, receiverId, isAI ? 'question_to_ai' : 'question_to_author']
      );

      // 7. Tạo question record
      const question = await queryRunner.query(
        `INSERT INTO questions
         (post_id, asker_id, target, content, fee, transaction_id,
          status, deadline_at)
         VALUES ($1, $2, $3, $4, 1000, $5, 'pending', NOW() + INTERVAL '48 hours')
         RETURNING *`,
        [postId, userId, dto.target, dto.content, transaction.id]
      );

      // 8. COMMIT
      await queryRunner.commitTransaction();

      // 9. Side effects (NGOÀI transaction)
      if (isAI) {
        // Đẩy vào queue để AI trả lời
        await this.aiQueue.add('answer-question', {
          questionId: question.id,
          postId,
          content: dto.content,
          authorId: post.authorId,
        });
      } else {
        // Gửi notification cho tác giả
        await this.notificationQueue.add('new-question', {
          authorId: post.authorId,
          questionId: question.id,
        });
      }

      return question;

    } catch (error) {
      await queryRunner.rollbackTransaction();
      throw error;
    } finally {
      await queryRunner.release();
    }
  }

  // jobs/refund.processor.ts
  @Processor('refund')
  export class RefundProcessor {

    @Cron('*/5 * * * *')  // Chạy mỗi 5 phút
    async processExpiredQuestions() {
      const queryRunner = this.dataSource.createQueryRunner();
      await queryRunner.connect();

      // Tìm tất cả question quá 48h chưa trả lời
      const expired = await queryRunner.query(`
        SELECT q.*, t.sender_id
        FROM questions q
        JOIN transactions t ON q.transaction_id = t.id
        WHERE q.status = 'pending'
          AND q.target = 'author'
          AND q.deadline_at < NOW()
        FOR UPDATE SKIP LOCKED
        LIMIT 100
      `);

      for (const question of expired) {
        await queryRunner.startTransaction();
        try {
          // Hoàn tiền về user
          await queryRunner.query(
            `UPDATE wallets SET balance = balance + $1 WHERE user_id = $2`,
            [question.fee, question.sender_id]
          );
          // Trừ tiền ở ví tác giả
          await queryRunner.query(
            `UPDATE wallets SET balance = balance - $1,
             total_earned = total_earned - $1
             WHERE user_id = $2`,
            [question.fee, question.answered_by ?? question.receiver_id]
          );
          // Update status
          await queryRunner.query(
            `UPDATE questions SET status = 'refunded', updated_at = NOW()
             WHERE id = $1`,
            [question.id]
          );
          // Ghi transaction hoàn tiền
          await queryRunner.query(
            `INSERT INTO transactions
             (sender_id, receiver_id, amount, type, status, reference_id, reference_type)
             VALUES ($1, $2, $3, 'refund', 'completed', $4, 'question')`,
            [SYSTEM_ADMIN_ID, question.sender_id, question.fee, question.id]
          );

          await queryRunner.commitTransaction();
        } catch (e) {
          await queryRunner.rollbackTransaction();
        }
      }
      await queryRunner.release();
    }
  }

  // ai/ai.service.ts
  async answerQuestion(questionId: string, postId: string,
                        content: string, authorId: string) {

    // 1. Embed câu hỏi
    const questionVector = await this.embeddingService.embed(content);

    // 2. Tìm kiếm vector tương tự trong bài viết + tài liệu ẩn
    const relevantChunks = await this.retrievalService.search({
      vector: questionVector,
      postId,
      authorId,
      topK: 5,
    });

    // 3. Build prompt
    const systemPrompt = `
      Bạn là trợ lý AI đại diện cho tác giả của bài viết này.
      Trả lời câu hỏi DỰA TRÊN các đoạn tài liệu được cung cấp.
      Nếu không tìm thấy thông tin liên quan, hãy nói rõ.
      Giọng điệu: thân thiện, chuyên nghiệp, như một chuyên gia.

      === TÀI LIỆU THAM KHẢO ===
      ${relevantChunks.map((c, i) => `[${i+1}] ${c.chunkText}`).join('\n\n')}
      === HẾT TÀI LIỆU ===
    `;

    // 4. Gọi LLM
    const answer = await this.geminiService.generate({
      systemPrompt,
      userMessage: content,
      temperature: 0.7,
      maxTokens: 1500,
    });

    // 5. Lưu câu trả lời
    await this.questionsRepo.update(questionId, {
      answer: answer.text,
      status: 'answered',
      answeredAt: new Date(),
    });

    return answer;
  }

  // ai/rag/retrieval.service.ts
  async search(params: { vector: number[], postId: string,
                          authorId: string, topK: number }) {

    // Tìm trong cả post_embeddings VÀ document_embeddings
    const results = await this.dataSource.query(`
      (
        SELECT chunk_text,
               1 - (embedding <=> $1::vector) AS similarity,
               'post' AS source
        FROM post_embeddings
        WHERE post_id = $2
      )
      UNION ALL
      (
        SELECT chunk_text,
               1 - (embedding <=> $1::vector) AS similarity,
               'document' AS source
        FROM document_embeddings
        WHERE author_id = $3
      )
      ORDER BY similarity DESC
      LIMIT $4
    `, [JSON.stringify(params.vector), params.postId,
        params.authorId, params.topK]);

    return results;
  }

  private detectQuestion(text: string): boolean {
    const questionPatterns = [
      /\?/g,
      /làm sao|như thế nào|tại sao|cho hỏi|giúp mình|giải thích/gi,
      /how|why|what|can you|please explain/gi,
    ];
    const matchCount = questionPatterns.filter(p => p.test(text)).length;
    return matchCount >= 2;
  }
